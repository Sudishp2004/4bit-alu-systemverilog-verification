class driver;
  virtual intf vif;
  mailbox gen2drv;
  function new (virtual intf vif, mailbox gen2drv);
    this.vif=vif;
    this.gen2drv=gen2drv;
  endfunction
  task main();
    repeat(2);
    begin
      transaction trans;
      gen2drv.get(trans);
      
      
      vif.A<=trans.A;
      vif.B<=trans.B;
      vif.opcode<=trans.opcode;
      #1;
      trans.display("display class signals");
      
    end
  endtask 
endclass
